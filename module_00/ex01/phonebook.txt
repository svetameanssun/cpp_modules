#include "PhoneBook.hpp"

class Contact{
	public:
		std::string first_name;
		std::string last_name;
		std::string nickname;
		std::string phone_number;
		std::string secret;
		void createContact() {
			std::cout << "Introduce name:\n";
			getline(std::cin, first_name);
			std::cout << "Introduce last name:\n";
			getline(std::cin, last_name);
			std::cout << "Introduce nickname:\n";
			getline(std::cin, nickname);
			std::cout << "Introduce phone number:\n";
			getline(std::cin, phone_number);
			std::cout << "Introduce your darkest secret:\n";
			getline(std::cin, secret);
		}
};

class PhoneBook{
	public:
		static const int maxCont = 2;
		Contact contacts[maxCont];
		void addContact(int *index, int *count) {
			if((*count) > maxCont)
			{
				if ((*index)== maxCont)
				{
					std::cout << "You reached the maximum number of contacts!\n";
					(*index) = 0;
					contacts[*index].createContact();
					(*index)++;
				}
				else
				{
					std::cout << "You will replce the existing contacts!\n";
					contacts[*index].createContact();
					(*index)++;
				}
			}
			else
			{
				contacts[*index].createContact();
				(*count)++;
				(*index)++;
			}
		}
		void displayPhoneBook()
		{
			int i = 0;
			while (i < maxCont) {
				std::cout << "Contact N " << i << "\n";
				std::cout << contacts[i].first_name << "\n";
				std::cout << contacts[i].last_name << "\n";
				std::cout << contacts[i].nickname << "\n";
				std::cout << contacts[i].phone_number << "\n";
				std::cout << contacts[i].secret << "\n";
				i++;
			}
		}
		void run(int *index, int *count) {
			std::string choice;
			std::cout << "CHOOSE:\nADD    SEARCH    EXIT\n";
			getline(std::cin, choice);
			if (choice == "EXIT") {
				exit(0);
			} else if (choice == "SEARCH") {
				displayPhoneBook();
			} else if (choice == "ADD") {
				addContact(index, count);
			} else {
				std::cout << "Unknown command.\n";
			}
		}
};

int main(int argc, char **argv)
{
	int count = 1;
	int index = 0;
	
	PhoneBook phonebookObj;
	while(1)
	{
		phonebookObj.run(&index, &count);
	}
}
